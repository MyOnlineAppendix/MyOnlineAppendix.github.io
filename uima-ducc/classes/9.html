  }

    e.printStackTrace();

  }
  URL[] urls=pcl.getURLs();
  if (urls.length != 1) {
    fail("Should have only one entry in the classpath, not " + urls.length);
  }
  boolean enable=false;
  if (enable) {
    File f=new File(privateCP[0]);
    URL u=f.toURI().toURL();
    if (!u.equals(urls[0])) {
      fail("Classpath should have " + u + " but instead has "+ urls[0]);
    }
  }
}
