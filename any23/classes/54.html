/** 
 * This test check if the RDFa1.1 CURIEs expansion is correct and backward
 * compatible with RDFa 1.0.
 * @throws Exception
 */
@Test public void testRDFa11CURIEs() throws Exception {
  assertExtract("/html/rdfa/rdfa-11-curies.html");
  assertModelNotEmpty();
  assertContains(RDFUtils.uri("http://dbpedia.org/resource/Albert_Einstein"),RDFUtils.uri("http://dbpedia.org/name"),RDFUtils.literal("Albert Einstein"));
  assertContains(RDFUtils.uri("http://dbpedia.org/resource/Albert_Einstein"),RDFUtils.uri("http://dbpedia.org/knows"),RDFUtils.uri("http://dbpedia.org/resource/Franklin_Roosevlet"));
  assertContains(RDFUtils.uri("http://database.org/table/Departments"),RDFUtils.uri("http://database.org/description"),RDFUtils.literal("Tables listing departments"));
  assertContains(RDFUtils.uri("http://database.org/table/Departments"),RDFUtils.uri("http://database.org/owner"),RDFUtils.uri("http://database.org/people/Davide_Palmisano"));
  assertContains(RDFUtils.uri("http://database.org/table/Departments"),RDFUtils.uri("http://xmlns.com/foaf/0.1/author"),RDFUtils.uri("http://database.org/people/Davide_Palmisano"));
  assertContains(RDFUtils.uri("http://database.org/table/Departments"),RDFUtils.uri("http://purl.org/dc/01/name"),RDFUtils.literal("Departments"));
  assertStatementsSize(null,null,null,6);
  logger.debug(dumpHumanReadableTriples());
}
