  byteBuffer.putInt(0);
  byteBuffer.putLong(virtualSize);
  byteBuffer.putInt(0);
  byteBuffer.putInt(0);
  byteBuffer.putLong(0L);
  byteBuffer.putLong(0L);
  byteBuffer.putInt(0);
  byteBuffer.putInt(0);
  byteBuffer.putLong(0L);
  inputStream=new ByteArrayInputStream(byteBuffer.array());
