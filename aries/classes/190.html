@Test public void testBundleDependencyUninstall() throws Exception {
  Subsystem root=getRootSubsystem();
  Subsystem subsystem=installSubsystemFromFile(APPLICATION_A);
  try {
    assertConstituent(root,BUNDLE_B);
    Bundle bundle=context(root).getBundleByName(BUNDLE_B);
    subsystem.uninstall();
    assertBundleState(bundle,Bundle.UNINSTALLED);
    assertNotConstituent(root,BUNDLE_B);
  }
  finally {
    if (!EnumSet.of(Subsystem.State.UNINSTALLING,Subsystem.State.UNINSTALLED).contains(subsystem.getState()))     uninstallSubsystemSilently(subsystem);
  }
}
