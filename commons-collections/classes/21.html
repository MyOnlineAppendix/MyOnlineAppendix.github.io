@Test public void testComparatorChainOnMinvaluedCompatator(){
  final ComparatorChain chain=new ComparatorChain();
  chain.addComparator(new Comparator(){
    public int compare(    final Integer a,    final Integer b){
      final int result=a.compareTo(b);
      if (result < 0) {
        return Integer.MIN_VALUE;
      }
      if (result > 0) {
        return Integer.MAX_VALUE;
      }
      return 0;
    }
  }
,true);
  assertTrue(chain.compare(Integer.valueOf(4),Integer.valueOf(5)) > 0);
  assertTrue(chain.compare(Integer.valueOf(5),Integer.valueOf(4)) < 0);

}
