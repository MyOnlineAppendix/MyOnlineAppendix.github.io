  assertTrue("False result for (true, true, true)",BooleanUtils.or(new boolean[]{true,true,true}));
  assertTrue("True result for (false, false)",!BooleanUtils.or(new boolean[]{false,false,false}));
  assertTrue("False result for (true, true, false)",BooleanUtils.or(new boolean[]{true,true,false}));
  assertTrue("False result for (true, false, true)",BooleanUtils.or(new boolean[]{true,false,true}));
  assertTrue("False result for (false, true, true)",BooleanUtils.or(new boolean[]{false,true,true}));
}
