@AfterClass public static void afterAll(){

        FileUtils.forceDelete(f);
      }
 catch (      IOException e) {
        e.printStackTrace();
      }
    }
  }
  if (container != null) {
    try {
      container.release(configurator);

 catch (    ComponentLifecycleException e) {
