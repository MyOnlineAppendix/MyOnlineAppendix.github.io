@Test public void testSample(){
  for (int i=0; i < 1000; i++) {
    Iterator t=new SamplingIterator(new CountingIterator(1000),0.1);
    int k=0;
    while (t.hasNext()) {
      int v=t.next();
      k++;
      assertTrue(v >= 0);
      assertTrue(v < 1000);
    }
    double sd=Math.sqrt(0.9 * 0.1 * 1000);
    assertTrue(k >= 100 - 4 * sd);
    assertTrue(k <= 100 + 4 * sd);
  }
}
