  HashMap map=Maps.newHashMap();
  for (  Gram n : input) {
    Gram val=map.get(n);
    if (val != null) {
      val.incrementFrequency(n.getFrequency());
    }
 else {
      map.put(n,n);
    }
  }
  int[] freq={5,3,9,5,6,7,8,1};

  for (int i=0; i < input.length; i++) {
