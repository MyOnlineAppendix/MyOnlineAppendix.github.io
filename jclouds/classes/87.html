@Test(dependsOnMethods="testListCookbookVersionsWithChefService") public void testDownloadCookbooks() throws Exception {
  Iterable cookbooks=chefService.listCookbookVersions();
  for (  CookbookVersion cookbook : cookbooks) {
    for (    Resource resource : ImmutableList.builder().addAll(cookbook.getDefinitions()).addAll(cookbook.getFiles()).addAll(cookbook.getLibraries()).addAll(cookbook.getSuppliers()).addAll(cookbook.getRecipes()).addAll(cookbook.getResources()).addAll(cookbook.getRootFiles()).addAll(cookbook.getTemplates()).addAll(cookbook.getAttributes()).build()) {

  }
