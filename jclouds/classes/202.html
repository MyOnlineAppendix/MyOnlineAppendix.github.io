@Test public void testSupplierExpirationReloadsTheCache(){
  ImageCacheSupplier imageCache=new ImageCacheSupplier(Suppliers.>ofInstance(images),3,Atomics.newReference(),Providers.of(getImageStrategy));
  assertEquals(imageCache.get().size(),1);
  Optional image=imageCache.get("foo");
  assertTrue(image.isPresent());
  assertEquals(image.get().getName(),"imageName-foo");
  assertEquals(imageCache.get().size(),2);
  Uninterruptibles.sleepUninterruptibly(3,TimeUnit.SECONDS);
  assertEquals(imageCache.get().size(),1);
  assertFalse(any(imageCache.get(),idEquals("foo")));
}
