@Test public void testSerializeCommitted() throws Exception {

  context.commitChanges();
  assertEquals(PersistenceState.COMMITTED,artist.getPersistenceState());
  ObjectId id=artist.getObjectId();
  assertNotNull(id);
  Artist deserialized=Util.cloneViaSerialization(artist);
  assertEquals("Unexpected persistence state: " + PersistenceState.persistenceStateName(deserialized.getPersistenceState()),PersistenceState.HOLLOW,deserialized.getPersistenceState());

  assertNull(deserialized.getObjectContext());
  assertEquals(null,deserialized.getArtistName());
}
