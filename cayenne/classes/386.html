@Test public void testSortDbEntities(){
  EntityResolver resolver=runtime.getDataDomain().getEntityResolver();
  List eSorted=Arrays.asList(resolver.getDbEntity("SORT_DEP"),resolver.getDbEntity("SORT_ROOT"));
  List e1=Arrays.asList(resolver.getDbEntity("SORT_ROOT"),resolver.getDbEntity("SORT_DEP"));
  List e2=Arrays.asList(resolver.getDbEntity("SORT_DEP"),resolver.getDbEntity("SORT_ROOT"));
  WeightedAshwoodEntitySorter sorter=new WeightedAshwoodEntitySorter();
  sorter.setEntityResolver(resolver);
  sorter.sortDbEntities(e1,false);
  assertEquals(eSorted,e1);
  sorter.sortDbEntities(e2,false);
  assertEquals(eSorted,e2);
}
