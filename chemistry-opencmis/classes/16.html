  expected.put("a","b");
  expected.put("c","d");
  assertEquals(expected,params);
  params.clear();
  expected.clear();
  assertEquals("inline",decodeContentDisposition("  inline ; a = \"b b\" (this is a comment) ; c =d;",params));
  expected.put("a","b b");
  expected.put("c","d");
  assertEquals(expected,params);
  params.clear();
  expected.clear();
  assertEquals("inline",decodeContentDisposition("inline; modification-date=\"Wed, 12 Feb 1997 16:29:51 -0500\"",params));
  assertEquals(Collections.singletonMap("modification-date","Wed, 12 Feb 1997 16:29:51 -0500"),params);
  params.clear();
}
