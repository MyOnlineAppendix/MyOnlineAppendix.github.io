@Test public void testUnsafeIntArray() throws Exception {
class IntArrayTest {
    private int[] array=new int[10];
  }
  Field field=IntArrayTest.class.getDeclaredField("array");
  ArrayPropertyAccessor propertyAccessor=(ArrayPropertyAccessor)new ReflectionPropertyAccessorFactory().fieldAccess(field,IntArrayTest.class);
  IntArrayTest test=new IntArrayTest();
  for (int i=0; i < 10; i++) {
    test.array[i]=i + 1;
  }
  IntArrayTest result=new IntArrayTest();
  for (int i=0; i < 10; i++) {
    int value=propertyAccessor.readInt(test.array,i);
    propertyAccessor.writeInt(result.array,i,value);
  }
  Arrays.equals(test.array,result.array);
}
