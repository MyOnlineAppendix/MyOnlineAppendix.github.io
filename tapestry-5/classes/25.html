@Test public void map_to_zipped_flow(){
  Map map=new HashMap();
  map.put(1,"fred");
  map.put(2,"barney");
  ZippedFlow zipped=F.zippedFlow(map);
  ZippedFlow sorted=zipped.sort(F.orderBySecond());
  assertFlowValues(sorted.firsts(),2,1);
  assertFlowValues(sorted.seconds(),"barney","fred");
  sorted=zipped.sort(F.orderByFirst());
  assertFlowValues(sorted.firsts(),1,2);
  assertFlowValues(sorted.seconds(),"fred","barney");
}
