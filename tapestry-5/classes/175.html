  assertText("//span[@id='persistedEntityClassName']",User.class.getName());
  clickAndWait("link=persist entity");
  assertText("//span[@id='name']","name");
  assertText("//span[@id='persistedEntityClassName']",PersistedEntity.class.getName());

  assertText("//span[@id='name']","name");
  assertText("//span[@id='persistedEntityClassName']",PersistedEntity.class.getName());
