  assertFieldValue("value1","");
  click("saveform");
  assertTextPresent("You must provide a value for Value1.");
  switchTabs(2);
  clickAndWait("saveform");
  assertTextPresent("You must provide a value for Value1.");
  switchTabs(1);
  click("frag1check");

  clickAndWait("saveform");
  assertFalse(isTextPresent("You must provide a value for Value1."));

  clickAndWait("saveform");

}
