Logger ExecutionTester 

@Test public void testTimeQueries() throws Exception {
  int i=0;
  long startTime=System.nanoTime();
  while (i < 1000) {
    assertQ(req("qt","/mcf","q","*:*","fl","id","AuthenticatedUserName","user1"),"//*[@numFound='3000']");
    assertQ(req("qt","/mcf","q","*:*","fl","id","AuthenticatedUserName","user2"),"//*[@numFound='3000']");
    assertQ(req("qt","/mcf","q","*:*","fl","id","AuthenticatedUserName","user3"),"//*[@numFound='2000']");
    i++;
  }
  System.out.println("Query time (milliseconds) = " + TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - startTime));
}
