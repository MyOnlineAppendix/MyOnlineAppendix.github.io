@Test public void testExclusions() throws PMDException {
  final AbstractFlexRule rule=new EmptyIfStmtRule();
  final IFlexFile duaneMxml=getTestFiles().get("bug.Duane.mxml");
  final Set excludes=new HashSet();
  excludes.add("");
  final int noExclusionViolationsLength=rule.processFile(duaneMxml,NodeFactory.createPackage(FileSetUtils.buildAst(duaneMxml)),getTestFiles()).size();
  rule.setExcludes(excludes);
  final int exclusionViolationsLength=rule.processFile(duaneMxml,NodeFactory.createPackage(FileSetUtils.buildAst(duaneMxml)),getTestFiles()).size();
  Assert.assertTrue(noExclusionViolationsLength > exclusionViolationsLength);
}
