        assertFalse(path,cppTestUser.hasPrivileges(tree,defaultPrivs.toArray(new String[defaultPrivs.size()])));
      }
    }
 else {
      assertFalse(path,cppTestUser.hasPrivileges(tree,REP_READ_NODES));
