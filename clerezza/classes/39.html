@Test public void testFilter() throws IOException {
  String fileName="filter.rdf";
  FileGraph mGraph=new FileGraph(new IRI(getTempFileUri(fileName)),Parser.getInstance(),Serializer.getInstance());
  mGraph.add(new TripleImpl(uriA,uriB,uriC));
  mGraph.add(new TripleImpl(uriC,uriB,uriA));
  mGraph=new FileGraph(new IRI(getTempFileUri(fileName)),Parser.getInstance(),Serializer.getInstance());
  assertEquals(2,mGraph.size());
  Iterator iterator=mGraph.filter(null,null,null);
  iterator.next();
  iterator.remove();
  assertEquals(1,mGraph.size());
  mGraph=new FileGraph(new IRI(getTempFileUri(fileName)),Parser.getInstance(),Serializer.getInstance());
  assertEquals(1,mGraph.size());
  deleteTempFile(fileName);
}
