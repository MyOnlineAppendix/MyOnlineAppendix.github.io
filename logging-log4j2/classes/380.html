@Test public void testGetCurrentStackTrace() throws Exception {
  final Stack> classes=ReflectionUtil.getCurrentStackTrace();
  final Stack> reversed=new Stack<>();
  reversed.ensureCapacity(classes.size());
  while (!classes.empty()) {
    reversed.push(classes.pop());
  }
  while (reversed.peek() != ReflectionUtil.class) {
    reversed.pop();
  }
  reversed.pop();
  assertSame(ReflectionUtilTest.class,reversed.pop());
}
