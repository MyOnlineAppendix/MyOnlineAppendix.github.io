/** 
 * Validates that the code pattern we use to add an appender on the fly
 * works with a basic appender that is not the new OutputStream appender or
 * new Writer appender.
 */
@Test public void testUpdatePatternWithFileAppender(){
  final LoggerContext ctx=(LoggerContext)LogManager.getContext(false);
  final Configuration config=ctx.getConfiguration();
  final Layout layout=PatternLayout.createDefaultLayout();
  final Appender appender=FileAppender.createAppender("target/" + getClass().getName() + ".log","false","false","File","true","false","false","4000",layout,null,"false",null,config);
  appender.start();
  config.addAppender(appender);
  ConfigurationTestUtils.updateLoggers(appender,config);
  LogManager.getLogger().error("FOO MSG");
}
