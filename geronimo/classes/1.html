@Test public void AsynEJBTest() throws Exception {

  httpMethod=new PostMethod(url);

  Assert.assertEquals(200,status);
  String result1=null;
  String result2=null;

    String response=new String(httpMethod.getResponseBodyAsString().getBytes("8859_1"));
