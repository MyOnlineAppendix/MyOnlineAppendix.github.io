  app.addAbsoluteOrdering(WebXml.ORDER_OTHERS);
  app.addAbsoluteOrdering("d");
  Set others=new HashSet();
  others.add(a);
  others.add(c);
  others.add(e);
  others.add(f);
  Set ordered=WebXml.orderWebFragments(app,fragments,null);

}
