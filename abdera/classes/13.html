@Test public void testUpdateEntry(){

  Document doc=resp.getDocument();

  entry.setTitle("This is the modified title");
  resp.release();

  assertEquals(ResponseType.SUCCESS,resp.getType());
  assertEquals(200,resp.getStatus());
