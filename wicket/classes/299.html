@Test public void isFirstLastPage(){

  pagecount=0;
  currentpage=0;
  assertTrue(link.isFirst());
  assertTrue(link.isLast());
  pagecount=1;
  currentpage=0;
  assertTrue(link.isFirst());
  assertTrue(link.isLast());
  pagecount=2;
  currentpage=0;

  currentpage=1;
