@Test public void testPropagationAllBookmarkable(){
  configure(new CdiConfiguration().setPropagation(ConversationPropagation.ALL));
  tester.startPage(TestConversationPage.class,new PageParameters().add("pageType","bookmarkable"));
  int i;
  for (i=0; i < 3; i++) {
    tester.assertCount(i);
    tester.clickLink("increment");
  }
  tester.clickLink("next");
  for (; i < 6; i++) {
    tester.assertCount(i);
    tester.clickLink("increment");
  }
}
