@Test public void testProcessInput() throws ExecException {
  Result res=op.processInput();
  assertEquals(POStatus.STATUS_EOP,res.returnStatus);
  op.attachInput(t);
  res=op.processInput();
  assertEquals(POStatus.STATUS_OK,res.returnStatus);
  assertEquals(t,res.result);
  op.detachInput();
  res=op.processInput();
  assertEquals(POStatus.STATUS_EOP,res.returnStatus);
  List inp=new ArrayList();
  inp.add(inpOp);
  op.setInputs(inp);
  op.processInput();
  assertEquals(POStatus.STATUS_EOP,res.returnStatus);
  inpOp.attachInput(t);
  res=op.processInput();
  assertEquals(POStatus.STATUS_OK,res.returnStatus);
  assertEquals(t,res.result);
  inpOp.detachInput();
  res=op.processInput();
  assertEquals(POStatus.STATUS_EOP,res.returnStatus);
}
