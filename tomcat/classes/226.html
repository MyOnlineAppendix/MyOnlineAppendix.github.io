  Iterator groupIterator=propertyGroups.iterator();
  JspPropertyGroupDescriptor groupDescriptor;

  Collection urlPatterns=groupDescriptor.getUrlPatterns();
  Assert.assertEquals(2,urlPatterns.size());
  Iterator iterator=urlPatterns.iterator();

  Assert.assertEquals("/bug49nnn/bug49726b.jsp",iterator.next());
