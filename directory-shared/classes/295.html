  String result=null;
  input="akarasulu**";
  result=Strings.trimConsecutiveToOne(input,'*');
  assertEquals("akarasulu*",result);
  input="*****akarasulu**";
  result=Strings.trimConsecutiveToOne(input,'*');
  assertEquals("*akarasulu*",result);
  input="**akarasulu";
  result=Strings.trimConsecutiveToOne(input,'*');
  assertEquals("*akarasulu",result);
  input="**akar****asulu**";
  result=Strings.trimConsecutiveToOne(input,'*');
  assertEquals("*akar*asulu*",result);
  input="akarasulu";
  result=Strings.trimConsecutiveToOne(input,'*');
  assertEquals("akarasulu",result);

  result=Strings.trimConsecutiveToOne(input,'*');
  assertEquals("*a*k*a*r*a*s*u*l*u*",result);
}
