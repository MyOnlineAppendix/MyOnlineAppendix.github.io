@Test public void testOverrideMethod() throws Exception {
  Set beanFields=JavaIntrospectionHelper.getAllUniquePublicProtectedMethods(Bean1.class,true);
  boolean invoked=false;
  for (  Method method : beanFields) {
    if (method.getName().equals("override")) {
      method.invoke(new Bean1(),"foo");
      invoked=true;
    }
  }
  if (!invoked) {
    throw new Exception("Override never invoked");
  }
}
