@Test public void testDetectText() throws Exception {
  assertText("Hello, World!".getBytes(UTF_8));
  assertText(" \t\r\n".getBytes(UTF_8));
  assertNotText(new byte[]{-1,-2,-3,0x09,0x0A,0x0C,0x0D,0x1B});
  assertNotText(new byte[]{0});
  assertNotText(new byte[]{'H','e','l','l','o',0});
  byte[] data=new byte[512];
  Arrays.fill(data,(byte)'.');
  assertText(data);
  Arrays.fill(data,100,110,(byte)0x1f);
  assertText(data);
  Arrays.fill(data,100,111,(byte)0x1f);
  assertNotText(data);
  Arrays.fill(data,(byte)0x1f);
  assertNotText(data);
  data=new byte[513];
  Arrays.fill(data,(byte)'.');
  data[0]=0x1f;
  assertText(data);
  Arrays.fill(data,100,150,(byte)0x83);
  assertText(data);
  Arrays.fill(data,100,200,(byte)0x83);
  assertNotText(data);
  Arrays.fill(data,(byte)0x1f);
  assertNotText(data);
}
