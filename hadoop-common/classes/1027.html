@Test public void testProxyServer(){

  assertFalse(ProxyServers.isProxyServer("1.1.1.1"));
  conf.set(ProxyServers.CONF_HADOOP_PROXYSERVERS,"2.2.2.2, 3.3.3.3");

  assertFalse(ProxyServers.isProxyServer("1.1.1.1"));

  assertTrue(ProxyServers.isProxyServer("3.3.3.3"));
