  createFile(src);
  createFile(dst);
  final FSNamesystem namesystem=cluster.getNamesystem();
  final long blocks=namesystem.getBlocksTotal();

  dst=getTestPath("testDeletionOfDstFile/newdir/dst");

      exists(fc,src);
      exception=false;
    }
 catch (    Exception e) {

      count++;
    }
  }

  Assert.assertTrue(exists(fc,dst));
