      targ=Integer.valueOf(random.nextInt(nObjects));
    }

      threads[i]=new InvalidateThread(pool,key,obj[targ.intValue()]);
    }
    for (int i=0; i < nThreads; i++) {
      new Thread(threads[i]).start();
    }
    boolean done=false;
    while (!done) {
      done=true;
      for (int i=0; i < nThreads; i++) {
        done=done && threads[i].complete();
