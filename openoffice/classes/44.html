@Test public void checkToolBoxItem() throws Exception {
  XModel aModel=(XModel)UnoRuntime.queryInterface(XModel.class,document);
  XController xController=aModel.getCurrentController();
  XText text=document.getText();
  text.setString("normal");
  XTextRange end=text.getEnd();
  end.setString("bold");
  UnoRuntime.queryInterface(XPropertySet.class,end).setPropertyValue("CharWeight",FontWeight.BOLD);
  UnoRuntime.queryInterface(XTextViewCursorSupplier.class,xController).getViewCursor().gotoRange(text,false);
  XInterface oObj=null;
  AccessibilityTools at=new AccessibilityTools();
  XWindow xWindow=at.getCurrentContainerWindow(getFactory(),aModel);
  XAccessible xRoot=at.getAccessibleObject(xWindow);
  oObj=at.getAccessibleObjectForRole(xRoot,AccessibleRole.TOGGLE_BUTTON,"Bold");
  assertNotNull("Found a TOGGLE_BUTTON",oObj);
  XAccessibleContext oContext=(XAccessibleContext)UnoRuntime.queryInterface(XAccessibleContext.class,oObj);
  XAccessibleStateSet oSet=oContext.getAccessibleStateSet();
  assertTrue("The 'INDETERMINATE' state is not set.",oSet.contains(AccessibleStateType.INDETERMINATE));
}
