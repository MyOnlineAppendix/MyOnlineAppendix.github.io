@Test public void testRepositoryManagerWeights() throws Exception {
  DefaultRepositoryManager repositoryManager=new DefaultRepositoryManager(null);
  RepoManager2 manager2=new RepoManager2();
  RepoManager1 manager1=new RepoManager1();
  List managers=new ArrayList();
  managers.add(repositoryManager);
  managers.add(manager2);
  managers.add(manager1);
  Collections.sort(managers,new RepositoryWeightComparitor());
  assertEquals(manager1,managers.get(0));
  assertEquals(manager2,managers.get(1));
  assertEquals(repositoryManager,managers.get(2));
}
