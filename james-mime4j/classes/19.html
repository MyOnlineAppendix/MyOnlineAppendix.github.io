@Test public void testGetParameter() throws Exception {

  f=parse("Content-Disposition: x-yada;" + "  fileNAme= \"ya:\\\"*da\"; " + "\tSIZE\t =  1234");

  Assert.assertEquals("\"hepp\"  =us\t-ascii",f.getParameter("x-yada"));
