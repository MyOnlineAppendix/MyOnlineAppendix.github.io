@Test public void testEscapedSoftBreak() throws Exception {
  byte[] content=new byte[500];
  Arrays.fill(content,(byte)0x20);
  byte[] expected=new byte[1557];
  int index=0;
  for (int l=0; l < 20; l++) {
    for (int i=0; i < 25; i++) {
      expected[index++]='=';
      expected[index++]='2';
      expected[index++]='0';
    }
    if (l < 19) {
      expected[index++]='=';
      expected[index++]='\r';
      expected[index++]='\n';
    }
  }
  check(content,expected);
}
