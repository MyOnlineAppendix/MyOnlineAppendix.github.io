@Test public void testConversationDialog() throws Exception {

  pageInteraction.checkTextValue("value1","1");
  pageInteraction.checkTextValue("value2","2");
  pageInteraction.checkTextValue("value3","3");
  pageInteraction.click("back").checkState(Pages.Conversation.Conversation3.class).useForm("conversation3");
  pageInteraction.click("conversation3:back").checkState(Pages.Conversation.Conversation2.class).useForm("conversation2");
  pageInteraction.setValue("conversation2:value2","new2").click("conversation2:nextPage").checkState(Pages.Conversation.Conversation3.class).useForm("conversation3");
  pageInteraction.setValue("conversation3:value3","new3").click("conversation3:submit").checkState(Pages.Conversation.Result.class);
  pageInteraction.checkTextValue("value1","1");
