@Test public void testMultiMapping_Webpage() throws IOException {
  MongoStore store=new MongoStore<>();
  store.setKeyClass(String.class);
  store.setPersistentClass(WebPage.class);
  MongoMappingBuilder builder=new MongoMappingBuilder<>(store);
  builder.fromFile("/multimapping.xml");
  MongoMapping mapping=builder.build();
