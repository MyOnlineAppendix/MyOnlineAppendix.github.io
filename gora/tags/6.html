 */
@Test public void testClear(){
  WebPage page=WebPage.newBuilder().build();
  page.setUrl(new Utf8("http://foo.com"));
  page.getParsedContent().add(new Utf8("foo"));
  page.getOutlinks().put(new Utf8("foo"),new Utf8("bar"));
