  for (  String s : printStackTraces(exceptionDefaultConstructor)) {
    assertStackTrace(s,"javax.ejb.EJBException","\tat");

    assertStackTrace(s,"javax.ejb.EJBException","\tat");

    assertStackTrace(s,"javax.ejb.EJBException","\tat");
  }
