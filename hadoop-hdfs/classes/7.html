@Test public void testEditsLogRename() throws Exception {

  Path dst1=getTestRootPath(fc,"testEditsLogRename/dstdir/dst1");
  createFile(src1);
  fs.mkdirs(dst1.getParent());
  createFile(dst1);
  fs.setQuota(dst1.getParent(),2,FSConstants.QUOTA_DONT_SET);
  fs.delete(dst1,true);

  restartCluster();
