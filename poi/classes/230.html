  cc2.setString(new XSSFRichTextString("A new comment"));

  workbook=XSSFTestDataSamples.writeOutAndReadBack(workbook);

  sheet2=workbook.getSheetAt(1);
  assertNotNull(sheet1.getRow(4).getCell(2).getCellComment());
