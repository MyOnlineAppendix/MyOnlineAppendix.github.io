@Test public void testOnTableWithMultipleEntries() throws Exception {
  for (int i=1; i < 10; i++) {
    String istr=Integer.toString(i);
    table.put(istr,istr);
  }
  Cursor>> cursor=new DupsContainerCursor(table);
  cursor.after(new Tuple>("2",null));
  assertTrue(cursor.next());
  Tuple> tuple=cursor.get();
  assertEquals("3",tuple.getKey());
  assertEquals("3",tuple.getValue().getArrayTree().getFirst());
  cursor.before(new Tuple>("7",null));
  cursor.next();
  tuple=cursor.get();
  assertEquals("7",tuple.getKey());
  assertEquals("7",tuple.getValue().getArrayTree().getFirst());
  cursor.last();
  cursor.next();
  assertFalse(cursor.available());
  cursor.beforeFirst();
  cursor.next();
  tuple=cursor.get();
  assertEquals("1",tuple.getKey());
  assertEquals("1",tuple.getValue().getArrayTree().getFirst());
  cursor.afterLast();
  assertFalse(cursor.next());
  cursor.beforeFirst();
  assertFalse(cursor.previous());

  cursor.after(new Tuple>("0",null));
