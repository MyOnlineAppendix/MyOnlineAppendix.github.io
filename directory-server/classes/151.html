  assertEquals(3,idx.count());

  IndexEntry e1=cursor.get();
  assertEquals(Strings.getUUID(555L),e1.getId());
  assertEquals("bar",e1.getKey());
  cursor.next();
  IndexEntry e2=cursor.get();
  assertEquals(Strings.getUUID(333L),e2.getId());

  cursor.next();

  cursor.close();
}
