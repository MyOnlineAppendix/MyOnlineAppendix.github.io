 */
@Test public void testAdminAccountCreation() throws Exception {
  String userDn="uid=admin,ou=system";
  LdapConnection connection=getConnectionAs(getService(),userDn,"secret");
  Entry entry=connection.lookup(userDn);
  performAdminAccountChecks(entry);
  assertTrue(ArrayUtils.isEquals(entry.get("userPassword").get().getBytes(),Strings.getBytesUtf8("secret")));
  connection.close();
  getService().shutdown();
  getService().startup();
  connection=getConnectionAs(getService(),userDn,"secret");
  entry=connection.lookup(userDn);
  performAdminAccountChecks(entry);
  assertTrue(ArrayUtils.isEquals(entry.get("userPassword").get().getBytes(),Strings.getBytesUtf8("secret")));
  connection.close();
}
