 * An attempt to use a closed connection must fail with the correct exception.
 * @throws SQLException if an error occurred while opening the database.
 */
@Test(expected=SQLConnectionClosedException.class) @DependsOnMethod("openCloseConnection") public void connectionClosed() throws SQLException {
  final Driver driver=new DBFDriver();
  final Connection connection=driver.connect(this.dbfFile.getAbsolutePath(),null);
  connection.close();
  try {
    connection.createStatement();
  }
 catch (  SQLConnectionClosedException e) {
    assertEquals("The database name in this exception is not well set.",e.getDatabase().getName(),this.dbfFile.getName());
    throw e;
  }
}

Class: org.apache.sis.internal.system.ReferenceQueueConsumerTest 

NullVerifier ExceptionVerifier HybridVerifier 

/** 
 * Verifies that invoking {@link Thread#interrupt()} will cause {@link InterruptedException}to be thrown even if invoked before {@link ReferenceQueue#remove()} put the
 * thread in a waiting state. This behavior is documented in {@link Object#wait()}, but
 * the reference queue javadoc is silent on this topic.
 * 
This method is not a test of the SIS library, but rather a verification of our JDK
 * library interpretation.

 * @throws InterruptedException This is the excepted exception.
 */
@Test(expected=InterruptedException.class) public void verifyInterruptAssumption() throws InterruptedException {
  final ReferenceQueue queue=new ReferenceQueue();
  Thread.currentThread().interrupt();
  assertNull(queue.remove(1000));
}
