  cs=GRID;
  assertSame(cs,cs.forConvention(AxesConvention.NORMALIZED));
  cs=GEOCENTRIC;
  assertSame(cs,cs.forConvention(AxesConvention.NORMALIZED));
  cs=CARTESIAN_2D;
  assertSame(cs,cs.forConvention(AxesConvention.NORMALIZED));
  cs=CARTESIAN_3D;
  assertSame(cs,cs.forConvention(AxesConvention.NORMALIZED));
  cs=PROJECTED;

  cs=GEODETIC_2D;
