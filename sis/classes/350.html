/** 
 * Tests a record where all members are the same primitive type. This method performs tests
 * similar to {@link #testSetAll()}, {@link #testToString()} and {@link #testSerialization()}.
 */
@Test @DependsOnMethod({"testSetAll","testToString","testSerialization"}) public void testPrimitiveType(){
  final Map> members=new LinkedHashMap>(8);
  assertNull(members.put("latitude",Double.class));
  assertNull(members.put("longitude",Double.class));
  final DefaultRecord record=new DefaultRecord(SerializableRecordSchema.INSTANCE.createRecordType("MyRecordType",members));
  assertEquals("baseValueClass",Double.TYPE,record.definition.baseValueClass());
  assertArrayEquals("attributes.values",new Double[]{0.0,0.0},record.getAttributes().values().toArray());
  setAllAndCompare(record,-13.1639,-72.5468);
  assertMultilinesEquals("Record[“MyRecordType”] {\n" + "    latitude  : -13.1639\n" + "    longitude : -72.5468\n"+ "}\n",record.toString());
  assertNotSame(record,assertSerializedEquals(record));
