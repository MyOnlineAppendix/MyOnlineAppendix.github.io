/** 
 * Verify the value of the {@link Transliterator#SPACES} constant.
 */
@Test public void testSpacesConstant(){
  int code=0;
  for (char c=0; c < 32; c++) {
    if (Character.isWhitespace(c)) {
      code|=(1 << c);
    }
  }
  assertEquals(Transliterator.SPACES,code);
}
