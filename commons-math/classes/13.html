@Test public void testParametricValue(){

  final double hi=3;
  final Sigmoid f=new Sigmoid(lo,hi);
  final Sigmoid.Parametric g=new Sigmoid.Parametric();

  Assert.assertEquals(f.value(2),g.value(2,new double[]{lo,hi}),0);
}
