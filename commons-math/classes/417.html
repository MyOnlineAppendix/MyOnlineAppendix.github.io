    int p3=((Integer)sample[2]).intValue();

    int i1=p1 * p2 * p3;
    int i2=p1 * p2 * p4;

    Assert.assertEquals(gcd,ArithmeticUtils.gcd(i1,i2));
    long l1=i1;
    long l2=i2;
    Assert.assertEquals(gcd,ArithmeticUtils.gcd(l1,l2));
