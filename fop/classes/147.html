@Test public void testImages() throws Exception {
  parseConfig(createRenderer());
  assertEquals(false,conf.isColorImages());
  assertEquals(Integer.valueOf(8),conf.getBitsPerPixel());
  ImagesModeOptions mode=MODE_GRAYSCALE;
  parseConfig(createRenderer().startImages(mode).setModeAttribute(mode.getModeAttribute(),String.valueOf(1)).endImages());
  assertEquals(false,conf.isColorImages());
  assertEquals(Integer.valueOf(1),conf.getBitsPerPixel());
  mode=MODE_COLOR;
  parseConfig(createRenderer().startImages(mode).setModeAttribute(mode.getModeAttribute(),String.valueOf(false)).endImages());
  assertEquals(true,conf.isColorImages());
  assertEquals(false,conf.isCmykImagesSupported());
  parseConfig(createRenderer().startImages(mode).setModeAttribute(mode.getModeAttribute(),String.valueOf(true)).endImages());
  assertEquals(true,conf.isColorImages());
  assertEquals(true,conf.isCmykImagesSupported());
}
