  GenericArray forward=new GenericData.Array(10,schema);
  GenericArray backward=new GenericData.Array(10,schema);
  for (int i=0; i <= 9; i++) {
    forward.add(i);
  }
  for (int i=9; i >= 0; i--) {
    backward.add(i);
  }
  forward.reverse();
  assertTrue(forward.equals(backward));
