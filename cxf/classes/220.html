  r1.setName(BookStore.class.getName());
  r1.setPath("/store");
  UserOperation op=new UserOperation();
  op.setName("getDescription");
  op.setVerb("GET");
  r1.setOperations(Collections.singletonList(op));
  UserResource r2=new UserResource();
