  assertNull(cfg1.getAcknowledgementInterval());
  assertFalse(cfg1.isExponentialBackoff());

  rma.setBaseRetransmissionInterval(abri);

  cfg1=RMPolicyUtilities.intersect(rma,cfg0);

  assertFalse(cfg1.isExponentialBackoff());
