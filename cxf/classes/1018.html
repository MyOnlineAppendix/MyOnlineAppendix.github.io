@Test public void testIdentifierEquals(){

  Identifier id2=null;
  assertTrue(AbstractSequence.identifierEquals(id1,id2));
  ObjectFactory factory=new ObjectFactory();

  assertTrue(!AbstractSequence.identifierEquals(id1,id2));
  id2=factory.createIdentifier();
  id2.setValue("seq2");
  assertTrue(!AbstractSequence.identifierEquals(id1,id2));
  id2.setValue("seq1");
