  script+="\"text\" -> T2;";

  script+="DocumentAnnotation{-> T4};";
  script+="Document{-> T5};";
  CAS cas=null;
  try {
    cas=RutaTestUtils.getCAS(document);
    Ruta.apply(cas,script);
  }
 catch (  Exception e) {

  RutaTestUtils.assertAnnotationsEquals(cas,5,1,"some text");
  cas.release();
}
