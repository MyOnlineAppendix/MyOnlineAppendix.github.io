@Test public void testArjUnarchive() throws Exception {
  StringBuilder expected=new StringBuilder();
  expected.append("test1.xml\n");
  expected.append("test2.xml\n");
  expected.append("\n");
  ArjArchiveInputStream in=new ArjArchiveInputStream(new FileInputStream(getFile("bla.arj")));
  ArjArchiveEntry entry;
  StringBuilder result=new StringBuilder();
  while ((entry=in.getNextEntry()) != null) {
    result.append(entry.getName());
    int tmp;
    while ((tmp=in.read()) != -1) {
      result.append((char)tmp);
    }
    assertFalse(entry.isDirectory());
  }
  in.close();
  assertEquals(result.toString(),expected.toString());
}
