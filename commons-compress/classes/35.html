    t.setSize(5);

    tout.closeArchiveEntry();

    tout.putArchiveEntry(t);
    tout.write(new byte[]{'!'});
    tout.closeArchiveEntry();
    tout.close();
    tout=null;
    tin=new TarArchiveInputStream(new FileInputStream(f));
    t=tin.getNextTarEntry();
    assertNotNull(t);

    assertNotNull(t);
