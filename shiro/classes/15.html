  p1=new WildcardPermission("one,two");

  assertTrue(p1.implies(p2));
  assertFalse(p2.implies(p1));
  p1=new WildcardPermission("one,two,three");
  p2=new WildcardPermission("one,three");
  assertTrue(p1.implies(p2));
