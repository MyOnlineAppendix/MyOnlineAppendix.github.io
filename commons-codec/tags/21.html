@Test public void testPhonemeComparedToLaterIsNegative(){
  for (  final Rule.Phoneme[] phs : makePhonemes()) {
    for (int i=0; i < phs.length; i++) {
      for (int j=i + 1; j < phs.length; j++) {
        final int c=Rule.Phoneme.COMPARATOR.compare(phs[i],phs[j]);
        assertThat("Comparing " + phs[i].getPhonemeText() + " to "+ phs[j].getPhonemeText()+ " should be negative",Integer.valueOf(c),new NegativeIntegerBaseMatcher());
      }
    }
  }
}
