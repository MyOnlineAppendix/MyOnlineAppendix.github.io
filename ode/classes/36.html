@Test(dataProvider="secured-services-bundles") public void executeProcess(String bundleName) throws Exception {
  if (server.isDeployed(new File(bundleName).getName())) {
    server.undeployProcess(bundleName);
  }
  server.deployProcess(bundleName);
  try {
    String response=sendRequestFile("http://localhost:8888/processes/helloWorld",bundleName,"testRequest.soap");
    System.out.println(response);
    assertTrue("'" + response + "' didn't contain expected content.",response.contains("helloResponse") && response.contains("Hello World"));
  }
  finally {
    server.undeployProcess(bundleName);
  }
}
