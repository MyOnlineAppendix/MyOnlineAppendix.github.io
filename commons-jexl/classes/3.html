@Test public void testLiteralWithNulls() throws Exception {

  Object[][] checks={{null,new Integer(10)},{new Integer(10),null},{new Integer(10),null,new Integer(10)},{"10",null},{null,"10",null}};
  JexlContext jc=new MapContext();
  for (int t=0; t < exprs.length; ++t) {
    JexlExpression e=JEXL.createExpression(exprs[t]);

    Assert.assertTrue(exprs[t],Arrays.equals(checks[t],(Object[])o));
  }
}
